### OBJECTIVE
Create a user site using Jekyll and GitHub Pages.


### JEKYLL INSTALLATION
ruby -v
gem -v
gem install bundler jekyll
jekyll -v

### GITHUB SETUP
# In GitHub, create a new public repository titled kattaoa.github.io

### JEKYLL SETUP
cd Desktop
jekyll new kattaoa.github.io
cd kattaoa.github.io
# if working on a project site as opposed to a user site: In _config.yaml, change baseurl to the name of the repo: kattaoa.github.io

### CONNECT GITHUB AND JEKYLL SETUPS
git init
git checkout -b master
git status
git add .
git commit -m "initial commit"
git remote add origin https://github.com/kattaoa/github.io.git
git push origin master

bundle exec jekyll serve # run for first time
# jekyll serve # run for second time and beyond; use '--draft' to see drafts

# CHANGE FRONT MATTER DEFAULTS
# nano _config.yml
# This is where you change the settings. We want to change the files in _posts to have the "post" layout with the author as "Omkar A. Katta"
defaults:
  -
    scope:
      path: ""		    # empty => all files
      type: "posts"   # applies to those in _posts
    values:
      layout: "post"
      author: "Omkar A. Katta"
  -
    scope:
      path: "dir1"    # applies to files in dir1      
    values:
      layout: "page"
